                      # 前端基础知识

                      # js 数据类型

                      # 1.基本数据类型

                      原始数据类型

                      javascript 中原始数据类型有 7 种,在内存中被保存在栈内存,并且只保存原始值,没有函数可调用。

                      7 种原始数据类型

                      • Number
                      • String
                      • undefined
                      • null
                      • Boolean
                      • bigInt
                      • symbol

                        BigInt 类型

                        BigInt是一种内置对象它提供了一种方法来表示大于 2 的 53 次方 - 1 的整数。这原本是 Javascript 中可以用 Number 表示的最大数字。BigInt 可以表示任意大的整数。

                        Symbol 类型

                        新的原始数据类型Symbol,表示独一无二的值;它的值是通过函数生成,不能使用 new;
                        let s = Symbol(); 或 let w = Symbol('w');

                      # 2.引用类型

                      引用类型

                      引用类型是Object 对象类型，在 javascript 中除了基本数据类型都是引用类型;
                      列如: 数组,对象,Date,function 等;
                      引用类型在内存中被保存在堆内存中;

                      # 3.判断数据类型的方式

                      • typeof

                        typeof

                        typeof 可以判断除了null之外的所有基本数据类型；
                        typeof null 由于历史 bug；返回 object；
                        对于引用类型，除了函数会判断成function,其他都会判断成object;

                      typeof 0; // number;
typeof 'hzg'; // string;
typeof undefined; // undefined;
typeof true; // boolean;
typeof 1n; // 1n表示bigint类型， bigint;
typeof Symbol('hzg'); // symbol;
typeof null; // object;

typeof function() {}; // function;
typeof new Object(); // object;
typeof []; // object;

                      • constructor

                        constructor

                        constructor 原理是它判断的是他构造函数的名；它也有局限性， undefined 和 null 判断不出来，因为他俩没有 constructor 属性，而且构造函数的名称是可以修改的；
                        备注：使用 object.create()创建的 js 对象，没有 constructor；

                        let a = [];
a.constructor === Array; // true;

let b = 123;
b.constructor === Number; // true;

let c = 'hgz';
c.constuctor === String; // true;

let d = {};
d.constructor === Object; // true;

                      • instanceof

                        instanceof

                        instanceof 只可以判断两个对象是否属于实例关系； 它是通过原型链判断对象的类型；判断不了基本数据类型；
                        instanceof 运算符是用来检测构造函数的 prototype 属性是否出现在某个实例对象的原型链上；

                      let a = [];
a instanceof Array; // true;

let b = new String('hzg');
b instanceof String; // true;

let e = 'hzg';
e instanceof String; //false

let c = true;

                      # 深浅拷贝

                      深浅拷贝

                      在 javascript 中对象是引用类型，保存的是地址。
                      浅拷贝：只是复制了指向某一对象的指针，新旧对象共享同一块内存；新对象某一属性的值一修改旧对象也同时修改；
                      深拷贝：会另外创建一个一模一样的对象，新旧对象不共享同一块内存；修改新对象不会影响旧对象；

                      # 数组的深浅拷贝

                      数组浅拷贝的几种方式

                      • 赋值

                      • (...)剩余运算符 针对数组中的基本数据类型深度拷贝，存在引用类型，就是浅拷贝

                      /* 深度克隆**/
function deepClone(obj) {
  //首先判断传递的参数数据类型，不是null；
  //根据传递的参数数据类型分别处理；1.当是数组或对象时做递归处理；2.当是普通值时直接赋值就OK啦；
  function isObj(o) {
    return (typeof o === 'function' || typeof o === 'object') && o != null;
  }

  if (!isObj(obj)) {
    throw Error('非对象!');
  }
  //判断是数组还是对象；
  let newObj = Array.isArray(obj) ? [...obj] : { ...obj };

  Reflect.ownKeys(newObj).forEach((key) => {
    newObj[key] = isObj(newObj[key]) ? deepClone(newObj[key]) : newObj[key];
  });

  return newObj;
}

                      # 变量声明提升

                      TIP

                      1.在javascript中，函数声明(function a(){})与变量声明(var)经常被javascript引擎隐士的提升到当前作用域的最顶部；
                      2.函数声明的优先级高于变量声明,如果变量名跟函数名相同且未赋值,则函数声明会覆盖变量声明;
                      3.声明语句中的赋值不会被提升,只是变量的声明被提升；

                      WARNING

                      es6 中(let ,const)不存在变量提升;
                      var 声明的挂载到window对象下,let ,const声明的挂载到全局变量下,es6之后window和全局变量不是一回事

                      # 作用域链

                      TIP

                      1). 函数在声明时就定义好了它的作用域;
                      2).函数内部的变量被使用时，首先会在自己的私有作用域下查找是否有这个变量,有就直接使用，没有就会向他的上一级查找，父级有就使用父级的，父级没有就会就会以此继续向上查找直到找到window为止，window有就使用没有就is not defined;这种查找机制我们叫做作用域链；

                      # 闭包

                      TIP

                      指函数在定义的词法作用域以外的地方被调用,闭包使得函数可以继续访问定义时的词法作用域;

                      闭包的几种表现形式

                      • 返回一个函数
                        function fn(){
    var a = 'javscript';
    function func(){
      console.log(a);
    }
    
    return func;
  }
  var func = fn();
  func();             //javscript;
                      • 作为函数参数传递
                        var a=1;
  function fn(){
    var a=2;
    function func(){
      console.log(a);
    }
    test(func);
  }
  function test(fu){
    //这就是闭包；
    fu();
  }

  fn();       // 输出2 ， 而不是 1；

                      # 原型与原型链

                      TIP

                      1.每一个函数都有一个prototype属性(显示原型),它默认指向一个空的Object对象(即:原型对象);
                      2.原型对象中有一个属性constructor,它指向构造函数对象;
                      3.每一个实例对象都有一个__proto__属性,为隐式原型;对象的隐式原型的值为其对应构造函数显式原型的值;
                      3.通过prototype属性可以给原型对象添加属性和方法,每一个实例化的对象都自动拥有原型中的属性(或方法);

                      原型例子

                        function FN(){
    this.name = '刘德华',
    this.age = 18
  }  
  FN.prototype.test = function(){
    console.log('hello test');
  }

  let fn = new Fn();
  console.log(fn.__proto__ == FN.prototype);      // true;

                      WARNING

                      大Object是所有对象的爸爸,所有对象都可以通过__proto__的方式找到它;
                      大Function是所有函数的爸爸,所有函数都可以通过__proto__找到它;

                        function Test(){
    this.name= 'uzi'
  }
  console.log(Test.constructor)           //f Function(){[native code]}
  
  let test = {};
  console.log(test.constructor)         //f Object(){[native code]}

                      # 继承

                      # 1.原型链继承

                      TIP

                      原理：子类的原型为父类的一个实例对象实现继承;

                      WARNING

                      缺陷：1.不能向父类构造函数传参；
                           2.父类上的引用类型属性会被所有实例所共享,其中一个实例改变时,会影响其他实例；

                        //父类
  function Person(){
    this.presonName = 'person';
  }
  Person.prototype.showPerson = function(){
    console.log(this.presonName);
  }

  //子类
  function Student(){
    this.studentName = 'student';
  }
 
  //1.子类的原型为父类的一个实例；2.让子类的原型的construct指向子类；
  Student.prototype = new Person();
  Student.prototype.consructor = Student;
  //给Student原型上添加属性和方法，要在原型链继承完之后添加，因为上面修改了prototype的指向为new Person对象；
  Student.prototype.showStudent = function(){
    console.log(this.studentName);
  }

  let person1 = new Person();
  let student1 = new Student();
  console.log(person1,student1,Student.prototype);
  //结果如下图所示;

                      # 2.借用构造函数实现继承

                      TIP

                      原理:假的继承方式,在子类的造函数中通过call的方式调用父类的构造函数

                      function Person(name,age){
  this.name = name,
  this.age = age
}

Person.Prototype.eat = function(){

function Student(name,age,salary){

}
let DaMing = new Student('DaMing',18,'10K');

                      # 3.组合继承

                      TIP

                      原理:原型链继承加构造函数继承一起完成；

                      WARNING

                      缺陷:并不是百分之百完美，父类构造函数被调用多次；

                        function Person(name,age){
    this.name = name;
    this.age = age;
  }
  Person.prototype.eat =function (){

  Studio.prototype = new Person();

                      # new的实现

                      TIP

                      思路:1.创建一个原始对象object；
                      2.通过arguments属性,把构造函数拿到;
                      3.把构造函数的显示原型指向新创建obj隐士原型上;
                      4.通过arguments把构造函数的参数拿到;
                      5.通过apply把参数传递给新创建的obj对象上;

                      function myNew(){
  let obj = new object();

  let _self = Array.from(arguments).sclice(0,1)[0];
  
  obj.__proto__ = _self.prototype;

  let arg = Array.from(arguments)[1];

  let newObj = _self.apply(obj,arg);

  return (typeof newObj === "object" )?newObj:obj;
}

使用:let testNew =  myNew(A,'a');

                      # 防抖debounce实现

                      TIP

                      原理:函数在单位时间内重复触发,时间清空从新计算；

                      function debunce(fn,deplay){
  var timer;
  return function (){
    var _this = this;
    var arg = arguments;

    if(timer){
        clearTimeout(timer);
      }
    timer = setTimeout(function(){
        fn.apply(_this,arg);
      },deplay);
  }

}

                      # 节流throttle实现

                      TIP

                      原理:当到达了一定的时间间隔就会执行一次，可以理解为缩减函数执行频率;

                      function throttle(fn,deplay){
    //通过常亮valid来标识确定是否有函数在执行；
  let valid = true;
  
  return function(){
    //如果有函数再执行，就return;
    if(!valid){
      return false;
    }

    valid = false;
    //通过valid来设置是否工作；
    setTimeout(()=>{
      fn(),
      valid = true;
    },deplay);
  }
}

                      # Promsie.all源码实现

                      TIP

                      原理:1.promise.all返回一个promise对象;
                      2.封装gen函数;

                      Promsie.prototype.all = function (promises){
  return new Promise((resolve,reject)=>{
    let done = gen(resolve,promises.length);

    promises.forEach((index,item)=>{
      item.then((data)=>{
          done(data,index);
      },reject);

    });

  });
};
function gen(resolve,length){
let count=0;
let arr=[];
return function(value,i){
  arr[i] = value;
  if(++count == length){
    resolve(arr);
  }
}
}
